def multi_step_method(y0, T, dt, a):
    t_values = np.arange(0, T + dt, dt)
    y_values = np.zeros(len(t_values))

    y_values[0] = y0
    y_values[1] = np.e**(dt**2)

    for i in range(2, len(t_values)):
        fn = 2 * t_values[i-1] * y_values[i-1]
        fn_minus_1 = 2 * t_values[i-2] * y_values[i-2]

        y_values[i] = (1+a)*y_values[i-1] - a*y_values[i-2] + (dt/2)*((3-a)*fn - (1+a)*fn_minus_1)

    return y_values, t_values
