def explicit_finite_difference(M, N, T):
    L = np.pi
    dt = T / N
    dx = np.pi / M
    alpha = dt / dx**2

    u = np.zeros((N + 1, M + 1))
    x = np.linspace(0, L, M + 1)
    for j in range(M+1):
        if x[j] >= 0 and x[j] <= L/2:
            u[0, j] = x[j]
        else:
            u[0, j] = L - x[j]

    for n in range(N):
        for j in range(1, M):
            u[n + 1, j] = u[n, j] + alpha * (u[n, j-1] - 2*u[n, j] + u[n, j+1])

    return x, u, alpha

  1. B). To implement the given method to solve for the heat equation, I will first use the Thomas algorithm to solve for the tridiagonal system we found in the lecture. I will then use that solver to calculate the u values at each x point as time increases.

def thomas_algorithm(a,b,c,d):
    n = len(d)
    c_star = np.zeros(n-1)
    d_star = np.zeros(n)
    
    c_star[0] = c[0] / b[0]
    d_star[0] = d[0] / b[0]
    
    for i in range(1, n-1):
        temp = b[i] - a[i-1] * c_star[i-1]
        c_star[i] = c[i] / temp
        d_star[i] = (d[i] - a[i-1] * d_star[i-1]) / temp
    
    d_star[-1] = (d[-1] - a[-1] * d_star[-2]) / (b[-1] - a[-1] * c_star[-2])
    
    x = np.zeros(n)
    x[-1] = d_star[-1]
    
    for i in range(n-2, -1, -1):
        x[i] = d_star[i] - c_star[i] * x[i+1]
    
    return x

def backward_time_difference(M, N, T):
    L = np.pi
    dt = T / N
    dx = np.pi / M
    alpha = dt / dx**2

    u = np.zeros((N + 1, M + 1))
    x = np.linspace(0, L, M + 1)
    for j in range(M+1):
        if x[j] >= 0 and x[j] <= L/2:
            u[0, j] = x[j]
        else:
            u[0, j] = L - x[j]

    a = -alpha * np.ones(M-2)
    b = (1 + 2*alpha) * np.ones(M-1)
    c = -alpha * np.ones(M-2)

    for n in range(N):
        u[n+1,1:-1] = thomas_algorithm(a, b, c, u[n,1:-1])
 
    return x, u, alpha
